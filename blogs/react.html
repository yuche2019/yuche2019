    src\ReactElement.js

    //https://gitee.com/mirrors/react/blob/v17.0.1/packages/react/src/ReactElement.js#L348
import { REACT_ELEMENT_TYPE } from "./ReactSymbols";
const RESERVED_PROPS = {
  key: true,
  ref: true,
  _store: true,
  __self: true,
  __source: true,
};
export function createElement(type, config, children) {
  const props = {};
  let key = null;
  if (config != null) {
    key = config.key;
  }
  for (let propName in config) {
    if (!RESERVED_PROPS.hasOwnProperty(propName)) {
      props[propName] = config[propName];
    }
  }
  const childrenLength = arguments.length - 2;
  if (childrenLength === 1) {
    props.children = children;
  } else if (childrenLength > 1) {
    const childArray = Array(childrenLength);
    for (let i = 0; i < childrenLength; i++) {
      childArray[i] = arguments[i + 2];
    }
    props.children = childArray;
  }

  const element = {
    $$typeof: REACT_ELEMENT_TYPE,
    type,
    key,
    props,
  };
  return element;
}

    # 3.3 优缺点

    # 3.3.1 优点

    • 处理了浏览器兼容性问题，避免用户操作真实 DOM，那么又麻烦又容易出错
    • 内容经过了 XSS 处理，可以防范 XSS 攻击

    # 3.3.2 缺点

    • 虚拟 DOM 需要消耗额外的内存
    • 首次渲染其实并不一定会更快

    # 4.3 相同点和不同点

    # 4.3.1 相同点

    • 它们都可以接收属性并且返回 React 元素

    # 4.3.2 不同点

    • 编程思想不同: 类组件需要创建实例，是基于面向对象的方式编程，而函数式组件不需要创建实例，接收输入，返回输出，是基于函数式编程的思路来编写的
    • 内存占用：类组件需要创建并保存实例，会占用一定内存，函数组件不需要创建实例，可以节约内存占用
    • 捕获特性：函数组件具有值捕获特性
    • 可测试性: 函数式组件更方便编写单元测试
    • 状态: 类组件有自己的实例，可以定义状态，而且可以修改状态更新组件，函数式组件以前没有状态，现在可以使用 useState 使用状态
    • 生命周期: 类组件有自己完整的生命周期，可以在生命周期内编写逻辑，函数组件以前没有生命周期，现在可以使用 useEffect 实现类似生命周期的功能
    • 逻辑复用: 类组件可以通过继承实现逻辑的复用，但官方推荐组件优于继承，函数组件可以通过自定义 Hooks 实现逻辑的复用
    • 跳过更新: 类组件可以通过shouldComponentUpdate和PureComponent来跳过更新，而函数式组件可以使用React.memo来跳过更新
    • 发展前景: 未来函数式组件将会成为主流，因为它可以更好的屏蔽 this 问题、规范和复用逻辑、更好的适合时间分片和并发渲染

    # 4.3.2.2 跳过更新
    class PureComponent extends Component {
  shouldComponentUpdate(newProps, nextState) {
    return (
      !shallowEqual(this.props, newProps) ||
      !shallowEqual(this.state, nextState)
    );
  }
}
function shallowEqual(obj1, obj2) {
  if (obj1 === obj2) {
    return true;
  }
  if (
    typeof obj1 != "object" ||
    obj1 === null ||
    typeof obj2 != "object" ||
    obj2 === null
  ) {
    return false;
  }
  let keys1 = Object.keys(obj1);
  let keys2 = Object.keys(obj2);
  if (keys1.length != keys2.length) {
    return false;
  }
  for (let key of keys1) {
    if (!obj2.hasOwnProperty(key) || obj1[key] !== obj2[key]) {
      return false;
    }
  }
  return true;
}

    # 5. 请说一下 React 中的渲染流程

    • 题目分析 本题属于原理题
    • 解题思路
      • 宏观的设计理念
      • 关键原理清晰描述，抽象和具象相结合
      • 结合工程实践和工作成果

    # 5.1 设计理念

    • 跨平台渲染=>虚拟 DOM
    • 快速响应=>异步可中断+增量更新

    # 5.2 性能瓶颈

    • JS 任务执行时间过长
      • 浏览器刷新频率为 60Hz,大概 16.6 毫秒渲染一次，而 JS 线程和渲染线程是互斥的，所以如果 JS 线程执行任务时间超过 16.6ms 的话，就会导致掉帧，导致卡顿，解决方案就是 React 利用空闲的时间进行更新，不影响渲染进行的渲染
      • 把一个耗时任务切分成一个个小任务，分布在每一帧里的方式就叫时间切片
    • 全量刷新 每次更新会全量重新渲染，性能较差。解决方案就是进行比较新旧的虚拟 DOM，找出差异，只更新变更部分的 DOM

    # 5.4.2 屏幕刷新率

    • 目前大多数设备的屏幕刷新率为 60 次/秒
    • 浏览器渲染动画或页面的每一帧的速率也需要跟设备屏幕的刷新率保持一致
    • 页面是一帧一帧绘制出来的，当每秒绘制的帧数（FPS）达到 60 时，页面是流畅的,小于这个值时，用户会感觉到卡顿
    • 每个帧的预算时间是 16.66 毫秒 (1 秒/60)
    • 1s 60 帧，所以每一帧分到的时间是 1000/60 ≈ 16 ms,所以我们书写代码时力求不让一帧的工作量超过 16ms

    # 5.4.3 帧

    • 每个帧的开头包括样式计算、布局和绘制
    • JavaScript 执行 Javascript 引擎和页面渲染引擎在同一个渲染线程,GUI 渲染和 Javascript 执行两者是互斥的
    • 如果某个任务执行时间过长，浏览器会推迟渲染

    # 5.4.4 requestIdleCallback

    • 我们希望快速响应用户，让用户觉得够快，不能阻塞用户的交互
    • requestIdleCallback 使开发者能够在主事件循环上执行后台和低优先级工作，而不会影响延迟关键事件，如动画和输入响应
    • 正常帧任务完成后没超过 16 ms,说明时间有富余，此时就会执行 requestIdleCallback 里注册的任务

    # 5.4.5 fiber

    • 我们可以通过某些调度策略合理分配 CPU 资源，从而提高用户的响应速度
    • 通过 Fiber 架构，让自己的调和过程变成可被中断。 适时地让出 CPU 执行权，除了可以让浏览器及时地响应用户的交互
    # 5.4.5.1 Fiber 是一个执行单元
    • Fiber 是一个执行单元,每次执行完一个执行单元, React 就会检查现在还剩多少时间，如果没有时间就将控制权让出去

    # 5.4.5.2 Fiber 是一种数据结构
    • React 目前的做法是使用链表, 每个 VirtualDOM 节点内部表示为一个 Fiber

    <!DOCTYPE html>
<html lang="en">
  <head>
    <meta charset="UTF-8" />

    <div id="root"></div>
    <script>
      let A1 = {
        type: "div",
        key: "A1",
        props: {
          children: [
            "A1文本",
            { type: "div", key: "B1", props: { children: "B1文本" } },
            { type: "div", key: "B2", props: { children: "B2文本" } },
          ],
        },
      };
      let rootFiber = {
        tag: "ROOT",
        key: "ROOT",
        stateNode: document.getElementById("root"),
        props: { children: [A1] },
      };

      //下一个工作单元
      let workInProgress = null;
      //render工作循环
      function workLoop(deadline) {
        while (deadline.timeRemaining() > 1 && workInProgress) {
          //执行一个任务并返回下一个任务
          workInProgress = performUnitOfWork(workInProgress);
        }
        if (workInProgress) {
          requestIdleCallback(workLoop);
        }
        //render阶段结束
      }
      function performUnitOfWork(fiber) {
        beginWork(fiber);
        if (fiber.child) {
          //如果子节点就返回第一个子节点
          return fiber.child;
        }
        while (fiber) {
          //如果没有子节点说明当前节点已经完成了渲染工作
          completeUnitOfWork(fiber); //可以结束此fiber的渲染了
          if (fiber.sibling) {
            //如果它有弟弟就返回弟弟
            return fiber.sibling;
          }
          fiber = fiber.return; //如果没有弟弟让爸爸完成，然后找叔叔
        }
      }
      function beginWork(fiber) {
        console.log("beginWork", fiber.key);
        if (fiber.props && fiber.props.children) {
          if (typeof fiber.props.children !== "string") {
            return reconcileChildren(fiber, fiber.props.children);
          }
        }
      }
      function reconcileChildren(returnFiber, newChildren) {
        let resultingFirstChild = null;
        let previousNewFiber = null;
        let newIdx = 0;
        for (; newIdx < newChildren.length; newIdx++) {
          let newFiber = createFiber(newChildren[newIdx]);
          newFiber.return = returnFiber;
          if (!previousNewFiber) {
            resultingFirstChild = newFiber;
          } else {
            previousNewFiber.sibling = newFiber;
          }
          previousNewFiber = newFiber;
        }
        returnFiber.child = resultingFirstChild;
        return resultingFirstChild;
      }
      function createFiber(element) {
        if (typeof element === "string") {
          return {
            tag: "TEXT",
            type: element.type,
            key: element,
            props: element,
          };
        } else {
          return {
            tag: "HOST",
            type: element.type,
            key: element.key,
            props: element.props,
          };
        }
      }
      function completeUnitOfWork(fiber) {
        console.log("completeUnitOfWork", fiber.key);
        let stateNode;
        if (fiber.tag === "ROOT") {
          stateNode = fiber.stateNode;
        } else if (fiber.tag === "TEXT") {
          stateNode = document.createTextNode(fiber.props);
          fiber.stateNode = stateNode;
        } else if (fiber.tag === "HOST") {
          stateNode = document.createElement(fiber.type);
          if (typeof fiber.props.children === "string") {
            stateNode.appendChild(
              document.createTextNode(fiber.props.children)
            );
          }
          fiber.stateNode = stateNode;
        }
        let child = fiber.child;
        while (!!child) {
          stateNode.appendChild(child.stateNode);
          child = child.sibling;
        }
      }
      workInProgress = rootFiber;
      requestIdleCallback(workLoop);
      /**
             beginWork ROOT
             beginWork A1
             beginWork A1文本
             completeUnitOfWork A1文本
             beginWork B1
             completeUnitOfWork B1
             beginWork B2
             completeUnitOfWork B2
             completeUnitOfWork A1
             completeUnitOfWork ROOT
             */
    </script>
  </body>
</html>

    import { createFiberRoot, createFiber } from "./ReactFiber";
import { HostRoot, HostComponent, HostText } from "./ReactWorkTags";
import { PerformedWork, Placement } from "./ReactFiberFlags";
import { REACT_ELEMENT_TYPE } from "./ReactSymbols";
import { setInitialProperties } from "./utils";
import { enqueueUpdate, createUpdate } from "./ReactUpdateQueue";

let workInProgress;
function render(element, container) {
  let fiberRootNode = createFiberRoot(container);
  updateContainer(element, fiberRootNode);
}

function updateContainer(element, fiberRootNode) {
  let current = fiberRootNode.current;
  let update = createUpdate();
  update.payload = { element };
  enqueueUpdate(current, update);
  scheduleUpdateOnFiber(fiberRootNode);
}

function scheduleUpdateOnFiber(fiberRootNode) {
  performSyncWorkOnRoot(fiberRootNode);
}

function performSyncWorkOnRoot(fiberRootNode) {
  workInProgress = createWorkInProgress(fiberRootNode.current);
  workLoopSync();
  commitRoot(fiberRootNode);
}

function commitRoot(fiberRootNode) {
  let finishedWork = fiberRootNode.current.alternate;
  let currentEffect = finishedWork.firstEffect;
  while (!!currentEffect) {
    let flags = currentEffect.flags;
    switch (flags) {
      case Placement: {
        commitPlacement(currentEffect);
        currentEffect.flags &= ~Placement;
        break;
      }
    }
    currentEffect = currentEffect.nextEffect;
  }
}
function getHostParentFiber(fiber) {
  var parent = fiber.return;
  while (!!parent) {
    if (isHostParent(parent)) {
      return parent;
    }
    parent = parent.return;
  }
}
function isHostParent(fiber) {
  return fiber.tag === HostComponent || fiber.tag === HostRoot;
}
function commitPlacement(node) {
  var parentFiber = getHostParentFiber(node);
  var parent;
  var parentStateNode = parentFiber.stateNode;
  switch (parentFiber.tag) {
    case HostRoot:
      parent = parentStateNode.container;
      break;
  }
  parent.appendChild(node.stateNode);
}
function workLoopSync() {
  while (!!workInProgress) {
    performUnitOfWork(workInProgress);
  }
}

function beginWork(current, workInProgress) {
  switch (workInProgress.tag) {
    case HostRoot:
      return updateHostRoot(current, workInProgress);
    case HostComponent:
      return updateHostComponent(current, workInProgress);
  }
}

function updateHostRoot(current, workInProgress) {
  let updateQueue = workInProgress.updateQueue;
  let nextChildren = updateQueue.pending.payload.element; //A1
  reconcileChildren(current, workInProgress, nextChildren);
  return workInProgress.child;
}

function updateHostComponent(current, workInProgress) {
  //如果是原生DOM节点
  let type = workInProgress.type;
  let nextProps = workInProgress.pendingProps;
  let nextChildren = nextProps.children;
  let isDirectTextChild = shouldSetTextContent(type, nextProps);
  if (isDirectTextChild) {
    nextChildren = null;
  }
  reconcileChildren(current, workInProgress, nextChildren);
  return workInProgress.child;
}

function shouldSetTextContent(type, props) {
  return (
    typeof props.children === "string" || typeof props.children === "number"
  );
}

let reconcileChildFibers = ChildReconciler(true);
let mountChildFibers = ChildReconciler(false);
function reconcileChildren(current, workInProgress, nextChildren) {
  if (!current) {
    //HostComponent A1
    workInProgress.child = mountChildFibers(workInProgress, null, nextChildren);
  } else {
    //HostRoot
    workInProgress.child = reconcileChildFibers(
      workInProgress,
      current.child,
      nextChildren
    );
  }
}
function ChildReconciler(shouldTrackSideEffects) {
  function placeSingleChild(newFiber) {
    if (shouldTrackSideEffects && !newFiber.alternate) {
      newFiber.flags = Placement;
    }
    return newFiber;
  }
  function reconcileChildFibers(returnFiber, currentFirstChild, newChild) {
    let isObject = typeof newChild === "object" && !!newChild;
    if (isObject) {
      switch (newChild.$$typeof) {
        case REACT_ELEMENT_TYPE:
          return placeSingleChild(
            reconcileSingleElement(returnFiber, currentFirstChild, newChild)
          );
      }
    }
    if (Array.isArray(newChild)) {
      return reconcileChildrenArray(returnFiber, currentFirstChild, newChild);
    }
  }
  return reconcileChildFibers;
}

function reconcileChildrenArray(returnFiber, currentFirstChild, newChildren) {
  let resultingFirstChild = null;
  let previousNewFiber = null;
  let newIdx = 0;
  for (; newIdx < newChildren.length; newIdx++) {
    let _newFiber = createChild(returnFiber, newChildren[newIdx]);
    placeChild(_newFiber, newIdx);
    if (!previousNewFiber) {
      resultingFirstChild = _newFiber;
    } else {
      previousNewFiber.sibling = _newFiber;
    }

    previousNewFiber = _newFiber;
  }

  return resultingFirstChild;
}
function placeChild(newFiber, newIndex) {
  newFiber.index = newIndex;
}
function createChild(returnFiber, newChild) {
  if (typeof newChild === "string" || typeof newChild === "number") {
    let created = createFiberFromText("" + newChild);
    created.return = returnFiber;
    return created;
  }

  if (typeof newChild === "object" && !!newChild) {
    switch (newChild.$$typeof) {
      case REACT_ELEMENT_TYPE: {
        let created = createFiberFromElement(newChild);
        created.return = returnFiber;
        return created;
      }
    }
    return null;
  }
}
function createFiberFromText(content) {
  let fiber = createFiber(HostText, content);
  return fiber;
}
/**
 * @param {*} returnFiber 父Fiber
 * @param {*} currentFirstChild 父Fiber的第一个儿子
 * @param {*} element 虚拟DOM的第一个元素
 * @returns 返回最的Fiber
 */
function reconcileSingleElement(returnFiber, currentFirstChild, element) {
  let created = createFiberFromElement(element);
  created.return = returnFiber;
  return created;
}
function createFiberFromElement(element) {
  let type = element.type;
  let key = element.key;
  let pendingProps = element.props;
  let fiberTag;
  if (typeof type === "string") {
    fiberTag = HostComponent;
  }
  let fiber = createFiber(fiberTag, pendingProps, key);
  fiber.type = type;
  return fiber;
}
function performUnitOfWork(unitOfWork) {
  let current = unitOfWork.alternate;
  let next = beginWork(current, unitOfWork);
  if (!next) {
    completeUnitOfWork(unitOfWork);
  } else {
    workInProgress = next;
  }
}
function completeUnitOfWork(unitOfWork) {
  //试图完成当前的工作单元，然后找弟弟，如果没有弟弟，就返回父fiber
  let completedWork = unitOfWork;
  do {
    let returnFiber = completedWork.return;
    completeWork(completedWork);
    if (!!returnFiber) {
      if (!returnFiber.firstEffect) {
        //如果父亲没有头，挂到父头上
        returnFiber.firstEffect = completedWork.firstEffect;
      }
      if (!!completedWork.lastEffect) {
        //如果自己有尾，而父亲也有尾,让父亲的尾巴下一个指向自己的头
        if (!!returnFiber.lastEffect) {
          returnFiber.lastEffect.nextEffect = completedWork.firstEffect;
        }
        returnFiber.lastEffect = completedWork.lastEffect; //父亲的尾指向自己的尾
      }
      var flags = completedWork.flags;
      if (flags > PerformedWork) {
        if (!!returnFiber.lastEffect) {
          //如果父亲有尾，尾巴下一个指向自己
          returnFiber.lastEffect.nextEffect = completedWork;
        } else {
          //如果父亲没有尾，父亲的头毛都指向自己
          returnFiber.firstEffect = completedWork;
        }
        returnFiber.lastEffect = completedWork;
      }
    }
    //先找弟弟
    var siblingFiber = completedWork.sibling;
    if (!!siblingFiber) {
      workInProgress = siblingFiber;
      return;
    }
    //再找父亲
    completedWork = returnFiber;
    workInProgress = completedWork;
  } while (!!completedWork); // 如果无弟，就完成父亲
}

function completeWork(workInProgress) {
  let newProps = workInProgress.pendingProps;
  switch (workInProgress.tag) {
    case HostRoot:
      return null;
    case HostComponent:
      let type = workInProgress.type;
      let instance = createInstance(type);
      appendAllChildren(instance, workInProgress);
      finalizeInitialChildren(instance, newProps);
      workInProgress.stateNode = instance;
      return null;
    case HostText:
      var newText = newProps;
      workInProgress.stateNode = createTextInstance(newText, workInProgress);
      return null;
  }
}
function createTextInstance(text) {
  var textNode = document.createTextNode(text);
  return textNode;
}
function createInstance(type) {
  let domElement = createElement(type);
  return domElement;
}
function finalizeInitialChildren(domElement, props) {
  setInitialProperties(domElement, props);
}

function appendAllChildren(parent, workInProgress) {
  let child = workInProgress.child;
  while (!!child) {
    if (child.tag === HostComponent || child.tag === HostText) {
      parent.appendChild(child.stateNode);
    }
    child = child.sibling;
  }
}

function createElement(type) {
  let domElement = document.createElement(type);
  return domElement;
}
function createWorkInProgress(current, pendingProps) {
  let workInProgress = current.alternate;
  if (!workInProgress) {
    workInProgress = createFiber(current.tag, pendingProps);
    workInProgress.type = current.type;
    workInProgress.stateNode = current.stateNode;
    workInProgress.alternate = current;
    current.alternate = workInProgress;
  }
  workInProgress.child = current.child;
  workInProgress.updateQueue = current.updateQueue;
  workInProgress.sibling = current.sibling;
  return workInProgress;
}

export default {
  render,
};
